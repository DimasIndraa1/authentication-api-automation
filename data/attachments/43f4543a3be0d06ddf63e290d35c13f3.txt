// variables 
const response = pm.response.json();
const accessToken_rs = response?.accessToken;
const refreshToken_rs = response?.refreshToken;
const id_rs = response?.id;
const username_rs = response?.username;
const email_rs = response?.email;
const firstName_rs = response?.firstName;
const lastName_rs = response?.lastName;
const gender_rs = response?.gender;
const image_rs = response?.image;
const nameRegex =  /^[a-zA-Z\s'-]+$/;
// variables environments
const username = pm.environment.get("username");
const password = pm.environment.get("password");
const expiresInMins = pm.environment.get("expiresInMins");
// Test
pm.test("01 | Status Code Valid yaitu 200", () => {
    pm.expect(pm.response.code).to.eql(200);
});
pm.test("02 | Memiliki Field Lengkap response dan tidak kosong", () => {
    pm.expect(response).to.not.be.null;
    pm.expect(response).to.include.all.keys("accessToken", "refreshToken", "id", "username", "email", "firstName", "lastName", "gender", "image").and.to.not.be.empty;
});
pm.test("03 | Password tidak ada di response", () => {
    pm.expect(response).to.not.have.property("password")
});
pm.test("04 | AccessToken tidak kosong dan valid menggunakan format JWT", () => {
    pm.expect(accessToken_rs).to.be.a("string").and.to.not.be.empty;
    pm.expect(accessToken_rs.split(".").every(part => part.length > 0)).to.be.true;
});
pm.test("05 | RefreshToken tidak kosong dan valid menggunakan format JWT", () => {
    pm.expect(refreshToken_rs).to.be.a("string").and.not.empty;
    pm.expect(refreshToken_rs.split(".").every(part => part.length > 0)).to.be.true;
});
pm.test("06 | Id bernilai angka positif (bukan 0 atau negatif)", () => {
    pm.expect(id_rs).to.be.a("number").and.to.be.above(0);
});
pm.test("07 | ID adalah bilangan bulat (integer)", () => {
    pm.expect(id_rs % 1).to.eql(0);
});
pm.test("08 | Username response sesuai dengan username request dan tidak ada spasi", () => {
    pm.expect(username_rs).to.equal(username);
    pm.expect(username_rs).to.not.include(" ");
});
pm.test("09 | Format pada email valid mengandung @ dan domain", () => {
    pm.expect(email_rs).to.include("@").and.to.match(/^[\w-\.]+@([\w-]+\.)+[\w-]{2,4}$/);
});
pm.test("10 | FirstName dan lastName hanya berisi karakter alfabet (tanpa angka/simbol)", () => {
    pm.expect(firstName_rs).to.be.a("string").and.to.match(nameRegex);
    pm.expect(lastName_rs).to.be.a("string").and.to.match(nameRegex);
});
pm.test("11 | FirstName & lastName tidak mengandung spasi di awal/akhir (trimmed)", () => {
    pm.expect(firstName_rs).to.eql(firstName_rs.trim());
    pm.expect(lastName_rs).to.eql(lastName_rs.trim());
});
pm.test("12 | Field gender hanya berisi value male / female", () => {
    pm.expect(gender_rs).to.be.oneOf(["male", "female"]);
});
pm.test("13 | Field image berupa URL valid dengan format (https://)", () => {
    pm.expect(image_rs).to.match(/^https?:\/\/.+/);
});
pm.test("14 | Image URL dapat diakses", () => {
    pm.sendRequest(image_rs, function (err, res) {
        pm.expect(err).to.be.null;
        pm.expect(res).to.have.property("code", 200);
    });
});
pm.test("15 | Response time di bawah 3 detik", () => {
    pm.expect(pm.response.responseTime).to.be.below(3000);
});
